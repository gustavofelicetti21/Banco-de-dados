A) QUAL A QUANTIDADE DE ANIMAIS REGISTRADOS NO SISTEMA?

B) QUAL O TOTAL DE VECINAS APLICADAS REGISTRADAS NO SISTEMA?

C) QUANTOS ANIMAIS TEMOS REGISTRADOS PARA CADA CATEGORIA?

D) QUAL A CATEGORIA DE ANIMAIS QUE MAIS VACINAS RECEBEU?

A)
SELECT COUNT(*)
FROM ANIMAIS;

B)
SELECT COUNT(*)
FROM VACINAS;

C)
SELECT t.NOME AS TIPO_ANIMAL, COUNT(a.ID) AS QTD_ANIMAIS
FROM TIPOS_ANIMAIS t
INNER JOIN ANIMAIS a ON t.ID = a.ID_TIPO_ANIMAL
GROUP BY t.NOME;

D)
SELECT t.NOME AS TIPO_ANIMAL, COUNT(v.ID) AS QTD_VACINAS
FROM TIPOS_ANIMAIS t
INNER JOIN ANIMAIS a ON t.ID = a.ID_TIPO_ANIMAL
INNER JOIN VACINAS v ON a.ID = v.ID_ANIMAL
GROUP BY t.NOME
ORDER BY QTD_VACINAS DESC
LIMIT 1;
